91 (
92  ADDRESS Ad,
93  PAGE *p
94 )
95 {
96  int cycle;
97  BYTE * pageTmp;
98  pageTmp = (BYTE*)p;
99 
101  Ad= Ad<<9;
102 
104  P3OUT &= ~BIT7;
105 
107  while((UCB0STAT & UCBUSY) == UCBUSY);
108  SPITXDATA = SPI_READ_4B_CMD;
109 
111  while((UCB0STAT & UCBUSY) == UCBUSY);
112  SPITXDATA = (BYTE)((Ad >> 24) & 0x000000FF);
113  while((UCB0STAT & UCBUSY) == UCBUSY);
114  SPITXDATA = (BYTE)((Ad >> 16) & 0x000000FF);
115  while((UCB0STAT & UCBUSY) == UCBUSY);
116  SPITXDATA = (BYTE)((Ad >> 8) & 0x000000FF);
117  while((UCB0STAT & UCBUSY) == UCBUSY);
118  SPITXDATA = (BYTE) (Ad & 0x000000FF);
119 
120  for(cycle = 0; cycle < PAGESIZE ; cycle++)
121  {
122  while((UCB0STAT & UCBUSY) == UCBUSY);
123  SPITXDATA = 0x00;
124 
125  *(pageTmp+cycle) = SPIRXDATA;
126  }
127 
129  while((UCB0STAT & UCBUSY) == UCBUSY);
130  P3OUT |= BIT7;
131 }
